punto 1:

SELECT p.Nombre, p.Precio, p.DIMENSIONES, e.Cantidad, pr.Nombre, pr.Domicilio, pr.Localidad
FROM `envios` e
JOIN productos p ON e.Codigo_Producto = p.Codigo_Producto
JOIN proovedores pr ON e.Numero_Proveedor = pr.Numero_Proveedor
ORDER BY p.Nombre ASC;

punto 3:
SELECT p.Nombre, p.Precio, p.DIMENSIONES, e.Cantidad, pr.Nombre, pr.Domicilio, pr.Localidad
FROM `envios` e
JOIN productos p ON e.Codigo_Producto = p.Codigo_Producto
JOIN proovedores pr ON e.Numero_Proveedor = pr.Numero_Proveedor
WHERE e.Cantidad >=200 && e.Cantidad<=300;

punto 7:
SELECT p.Nombre, p.Precio, p.DIMENSIONES, e.Cantidad, pr.Nombre, pr.Domicilio, pr.Localidad, ROUND((e.Cantidad * p.Precio), 2) AS Monto
FROM envios e
JOIN productos p ON e.Codigo_Producto = p.Codigo_Producto
JOIN proovedores pr ON e.Numero_Proveedor = pr.Numero_Proveedor;
